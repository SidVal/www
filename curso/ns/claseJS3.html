Tarea

Primero los enunciados de la tarea, más abajo el código para probar en la consola.

1. Calculadora (Parte 1)

Se le pedirá al usuario 2 números para generar la suma, la resta, la división y la multiplicación entre ellos (Preguntar al usuario que operación quiere realizar).

2. Par / Impar

Se le pedirá al usuario un número y se le mostrará en pantalla si es divisible por 2 o no, dando como resultado un número entero.

3. Tabla de Multiplicar hasta el 10

Se le pedirá al usuario un número y una vez que lo tengamos se deberá mostrar en pantalla las tablas de dicho número hasta el 10:
Por ejemplo, si el usuario escribe 5 se deberá generar lo siguiente:

  • 5 x 1 = 5
  • 5 x 2 = 10

Código de Tarea

// Ejercicios Clase 3
  // 1. Calculadora (Parte 1)
  // Se le pedirá al usuario 2 números para generar la suma, la resta, la división y la multiplicación entre ellos (Preguntar al usuario que operación quiere realizar).
  
  function operaciones() { //USÉ función para poder llamar al inicio del switch cuando valido lo que presiona el usuario
  
      let num1 = parseInt(prompt("Número 1:"));
      let num2 = parseInt(prompt("Número 2:"));
      let oper = prompt("¿Qué operación quiere realizar? \n Puedes elegir una letra:\n A = SUMA \n B = RESTA \n C = MULTIPLICACIÓN \n D = DIVISIÓN").toLowerCase();
  
      switch (oper) {
          case "a":
              let suma = num1 + num2;
              alert("El resultado de su SUMA es: " + suma);
              break;
          case "b":
              let resta = num1 - num2;
              alert("El resultado de su RESTA es: " + resta);
              break;
          case "c":
              let multi = num1 * num2;
              alert("El resultado de su MULTIPLICACION es: " + multi);
              break;
          case "d":
              let divi = num1 / num2;
              if (num2 == 0) {
                  alert("Por favor, no puedes dividir por cero. Vuelves al inicio, pon otro número.");
                  operaciones();
              } else {
                  alert("El resultado de su DIVISION es: " + divi);
              }
              break;
          default:
              alert("Por favor, prestar atención, no seleccionaste correctamente una letra de operación, por ejemplo, si quiere sumar, presione: A \n\n Operaciones:\n A = SUMA \n B = RESTA \n C = MULTIPLICACIÓN \n D = DIVISIÓN");
              alert("Empezas de cero ahora, de nuevo elegí números y operación!");
              operaciones();
              break;
      }
  }

// 2. Par / Impar
// Se le pedirá al usuario un número y se le mostrará en pantalla si es divisible por 2 o no, dando como resultado un número entero.

let num1 = parseInt(prompt("¿Qué número eliges?"));
if (num1 % 2 == 0) {
    alert("Tu número " + num1 + " es divisible por 2! ¡Excelente!");
} else {
    alert("Tu selección es un número impar, no da como resultado un número entero")
}

for (let i=1;i<11;i++) {
    document.write("
  • " + num1 + " x " + i + " = " + (num1*i) + "
